## fish shell使用
```
set fish_plugins rbenv bundler node brew

# 路径配置
set -x PATH ~/.nvm/versions/node/v6.9.2/bin /usr/local/bin /usr/bin /bin /usr/sbin /sbin /usr/local/go/bin ~/GoWork/bin $PATH
set -x GOPATH ~/GoWork $GOPATH

#alias
alias md "mkdir"
alias gst "git status"
alias gup "git pull --rebase"
alias gp "git push"
alias gstp "git stash pop"
```
